The following is an example SIMPL program to calculate the nth prime number.

prime(n)
        |i j current out|
        i <- 1!
        current <- 2!
        while (i > n) = 0?
                j <- 2!
                out <- 0!
                while ((j > current - 1) = 0) * (out = 0)?
                        if current / j * j = current?
                                out <- 1!
                        :!
                        j <- j + 1!
                !
                if out?:
                        i <- i + 1!
                !
                current <- current + 1!
        !
        prime <- current - 1!
!
